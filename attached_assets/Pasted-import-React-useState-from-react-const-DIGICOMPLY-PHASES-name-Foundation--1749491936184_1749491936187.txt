import React, { useState } from 'react';

const DIGICOMPLY_PHASES = [
  {
    name: 'Foundation',
    conditions: (data) => data.revenue < 10,
    coreServices: ['INC-20A', 'ADT-1', 'Startup India'],
    optional: ['MSME', 'GST Registration'],
  },
  {
    name: 'Stability',
    conditions: (data) => data.revenue >= 10 && data.revenue <= 100,
    coreServices: ['AOC-4', 'MGT-7', 'ITR-6'],
    optional: ['ISO Certification', 'Trademark'],
  },
  {
    name: 'Growth',
    conditions: (data) => data.revenue > 100,
    coreServices: ['ESOP Setup', '409A Valuation', 'Audit Support'],
    optional: ['Foreign Subsidiary Setup', 'FEMA Compliances'],
  }
];

export default function OnboardingPhaseEngine() {
  const [formData, setFormData] = useState({
    companyName: '',
    cin: '',
    revenue: '',
    servicesAvailed: []
  });

  const [result, setResult] = useState(null);

  const handleChange = (e) => {
    const { name, value } = e.target;
    setFormData((prev) => ({ ...prev, [name]: value }));
  };

  const handleSubmit = (e) => {
    e.preventDefault();
    const revenue = parseFloat(formData.revenue);
    const userData = { ...formData, revenue };
    const matchedPhase = DIGICOMPLY_PHASES.find(p => p.conditions(userData));
    setResult(matchedPhase);
  };

  return (
    <div className="max-w-3xl mx-auto p-6">
      <h1 className="text-2xl font-bold mb-4">DigiComply Phase Engine</h1>
      <form onSubmit={handleSubmit} className="space-y-4">
        <input name="companyName" placeholder="Company Name" className="w-full p-3 border rounded" onChange={handleChange} required />
        <input name="cin" placeholder="CIN" className="w-full p-3 border rounded" onChange={handleChange} required />
        <input name="revenue" type="number" placeholder="Annual Revenue (â‚¹ in Lakhs)" className="w-full p-3 border rounded" onChange={handleChange} required />
        <button type="submit" className="w-full bg-blue-600 text-white py-2 rounded hover:bg-blue-700">Detect Phase</button>
      </form>

      {result && (
        <div className="mt-8 bg-gray-100 p-4 rounded">
          <h2 className="text-xl font-semibold">You are in the <span className="text-blue-600">{result.name}</span> Phase</h2>
          <p className="mt-2">Core Services:</p>
          <ul className="list-disc list-inside text-sm text-gray-700">
            {result.coreServices.map(s => <li key={s}>{s}</li>)}
          </ul>
          <p className="mt-2">Optional Recommendations:</p>
          <ul className="list-disc list-inside text-sm text-gray-500">
            {result.optional.map(s => <li key={s}>{s}</li>)}
          </ul>
        </div>
      )}
    </div>
  );
}
